Problem 7: Prompt and Answer

Prompt: List all of the prime numbers from 1 to 1000
Answer:
prime<-function(x,y)
{
  a<-seq(x,y,1)
  for(i in a)
  {
    c<-"Prime"
    if(i<=1){ c<-"Not Prime"}
    if (i>3)
    {
      
      b<-floor(sqrt(i))
      for(j in 2:b)
      {
        
        if(i%%j==0)
        {
          c<-"Not Prime"
        }
      }
      
    }
    if(c=="Prime")
      {
      print(c(c,i))
      }
  }
}
