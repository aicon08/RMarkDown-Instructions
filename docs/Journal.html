Prompt: Write your own functions to give descriptive statistics for a vector variable storing multiple numbers. Write functions for the following without using R intrinsics: mean, mode, median, range, standard deviation
Mean Answer:
my_mean<-function(x)
{
  return(sum(x)/length(x))
  
}
t<-c(1:20)
my_mean(t)
## [1] 10.5
Mode Answer:
my_mode<-function(x)
{
 a<-sort(x)
  b<-unique(a)
  maxn<-0
  countn<-0
  for (i in b)
  {
  c<-a[a==i]
  if(length(c)>countn)
    {
    maxn=i
    countn=length(c)
    }
  }
 return(maxn)
}

my_mode(c(1,1,1,10,9,8,8,5,5,5,5))
## [1] 5
Median Answer:
my_median<-function(x)
{
  a<-sort(x)
  b<-length(x)
  c<-floor(b/2)
  if(b%%2==0)
  {
    return(mean(c(a[c],a[c+1])))
    
  }
  else
  {
   return(a[c+1])   
  }
}

my_median(c(1,2,3,4,5))
## [1] 3
my_median(1:10)
## [1] 5.5
Range Answer:
my_range<-function(x)
{
  sort(x)
  print(c(min(x),max(x)))
}

my_range(c(1,1,1,10,9,8,8,5,5,5,5))
## [1]  1 10
Standard Deviation Answer:
my_sd<-function(x)
{
  avgx<-mean(x)
  sd<-0
  for (i in x)
  {
    sd<-(i-avgx)^2 + sd
  }
  return(sqrt(sd/(length(x)-1)))
}
my_sd(c(-5,1,8,7,2))
## [1] 5.22494

Prompt: List all of the prime numbers from 1 to 1000
Answer:
prime<-function(x,y)
{
  a<-seq(x,y,1)
  for(i in a)
  {
    c<-"Prime"
    if(i<=1){ c<-"Not Prime"}
    if (i>3)
    {
      
      b<-floor(sqrt(i))
      for(j in 2:b)
      {
        
        if(i%%j==0)
        {
          c<-"Not Prime"
        }
      }
      
    }
    if(c=="Prime")
      {
      print(c(c,i))
      }
  }
}
